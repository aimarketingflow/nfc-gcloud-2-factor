Key Implementation: PBKDF2 Key Derivation
import hashlib
import os
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC

def derive_vault_key(nfc_uid: str, device_fingerprint: str) -> bytes:
    """Derive encryption key from NFC UID and device fingerprint"""
    
    # Convert inputs to bytes
    password = nfc_uid.encode('utf-8')
    salt = device_fingerprint.encode('utf-8')
    
    # PBKDF2-SHA256 with 100,000 iterations
    kdf = PBKDF2HMAC(
        algorithm=hashes.SHA256(),
        length=32,  # 256-bit key
        salt=salt,
        iterations=100000,
    )
    
    return kdf.derive(password)

def encrypt_vault(data: dict, key: bytes) -> bytes:
    """Encrypt credential vault using AES-256-GCM"""
    from cryptography.fernet import Fernet
    import base64
    
    # Create Fernet key from derived key
    fernet_key = base64.urlsafe_b64encode(key)
    fernet = Fernet(fernet_key)
    
    # Encrypt the serialized data
    import json
    plaintext = json.dumps(data).encode()
    ciphertext = fernet.encrypt(plaintext)
    
    return ciphertext
JWT Token Implementation
import jwt
import datetime
from typing import Dict, Any

def generate_jwt_token(device_id: str, ip_address: str, 
                      nfc_verified: bool = True) -> str:
    """Generate JWT token with security claims"""
    
    payload = {
        'iss': 'aimf-auth-server',  # Issuer
        'sub': device_id,           # Subject (device ID)
        'iat': datetime.datetime.utcnow(),  # Issued at
        'exp': datetime.datetime.utcnow() + datetime.timedelta(hours=8),
        'nfc_verified': nfc_verified,
        'device_id': device_id,
        'ip_binding': ip_address,
        'session_id': generate_session_id(),
        'auth_level': 'nfc_physical'
    }
    
    # Sign with RS256 (RSA-SHA256)
    token = jwt.encode(
        payload, 
        private_key, 
        algorithm='RS256'
    )
    
    return token

def validate_jwt_token(token: str, expected_ip: str) -> Dict[str, Any]:
    """Validate JWT token and claims"""
    try:
        # Decode and verify
        payload = jwt.decode(
            token, 
            public_key, 
            algorithms=['RS256']
        )
        
        # Validate IP binding
        if payload.get('ip_binding') != expected_ip:
            raise ValueError("IP address mismatch")
        
        # Validate NFC verification
        if not payload.get('nfc_verified'):
            raise ValueError("NFC verification required")
        
        return payload
        
    except jwt.ExpiredSignatureError:
        raise ValueError("Token expired")
    except jwt.InvalidTokenError as e:
        raise ValueError(f"Invalid token: {e}")

NFC Reader Not Detected

Symptom: "No NFC reader found" error

Solution:

  • Verify USB connection and driver installation
  • Check permissions: sudo usermod -a -G dialout $USER
  • Restart pcscd service: sudo systemctl restart pcscd

Vault Decryption Failed

Symptom: "Invalid vault key" error

Solution:

  • Ensure using the correct NFC token
  • Verify device fingerprint hasn't changed
  • Re-initialize vault if hardware changed

JWT Authentication Failed

Symptom: "Device not pre-authorized" error

Solution:

  • Re-register device with auth server
  • Check network connectivity to AIMF server
  • Verify system clock is synchronized
